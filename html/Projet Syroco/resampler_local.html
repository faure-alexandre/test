# -*- coding: utf-8 -*-

import time
tps_debut = time.perf_counter()

# Imports
import numpy as np
import pandas as pd
import os
import argparse
import sys
import json
from pprint import pprint

# Class imports
from VrsFiles import VrsFiles
from CaptorsFiles import CaptorsFiles
from ShipResampler import ShipResampler
from Logger import Logger
from Renamer import Renamer


def main():
    """ 
    Function which allows to execute the pre-processing script locally
    
    Parameters
    ----------
        
    Returns
    -------
    ship_resampler_object : ShipResampler
        the ShipResampler object containing the pre-processed data
    """

    parser = argparse.ArgumentParser(description='Apply resampling preprocessing script to onboard data')
    parser.add_argument('--version', action='version', version='%(prog)s 1.0')
    parser.add_argument('--verbose', '-v', action='store_true', help='print executions d√©tails in the prompt')
    
    # If the script is started from the command line, we get the parameters
    if sys.stdin.isatty(): 
        # Add of mandatory command line parameter
        parser.add_argument('json_params', metavar='Parameters_File', type=str,
                        help='Json file containing resampling parameters')
        
        # Get command line arguments
        args = parser.parse_args()
        # Opening the JSON file 
        params = open(args.json_params,) 
        # Deserailizing the data
        params = json.load(params)
        
    # Otherwise the parameters are hard written (exactly the same format as the json)
    else:
        params = \
            {
    "captors_files": {"C:/Users/Alex/Documents/Digital Lab/Projet 1/data/Onboard sensors/2021-01-12 16_19_30_CMA-CGM-Magellan.xlsx":
                       ["BonVoyage", "Bridge Sensor", "Motion Sensor", "NMEA GPS", "NAVIGATION_GPS"]},
    "vrs_files": ["C:/Users/Alex/Documents/Digital Lab/Projet 1/data/Onboard sensors/VRS MAGELLAN 2019.xlsx"],
    "name": "CMA CGM MAGELLAN",
    "selected_variables": {},
    "resampling_windows": "15min",
    "interpolation_method": "linear",
    "interpolation_limit_time": "12h",
    "NaNs_percentage": 0.2,
    "agregation_function": "np.nanmean",
    "set_custom_agg_function": {},
    "set_custom_inter_method": {"VRS": "nearest"},
    "set_custom_inter_limit_time": {"VRS": "1d"},
    "geospatial_format": "dm",
    "dist_limit": 50,
    "agregation_string_function": "first",
    "fill_method_str": "ffill",
    "deg_format": "initial",
    "same_sheets": [["NMEA Gyro","NAVIGATION_GYRO"], ["NMEA GPS","NAVIGATION_GPS"],
                       ["Bridge Sensor","Motion Sensor"]],
    "output_file": "C:/Users/Alex/Documents/GitHub/Projet Syroco/data/xlsx"
    }
        # Add verbose argument
        args = parser.parse_args(['--verbose'])


    # The variables are assigned (each key becomes a variable name and each value the content of the variable.
    globals().update(params)

    output_file = globals().get('output_file')
    
    # Default management of the output file name
    path_split = os.path.split(output_file)
    
    if path_split[1] == "csv":
        output_file = path_split[0] + '/' + str(name).replace(' ', '_') + '_' + str(resampling_windows) + '_' \
            + str(interpolation_method) + '_limit' + str(interpolation_limit_time) + '_nans' + str(NaNs_percentage) + '.csv'
    elif path_split[1] == "xlsx":
        output_file = path_split[0] + '/' + str(name).replace(' ', '_') + '_' + str(resampling_windows) + '_' \
            + str(interpolation_method) + '_limit' + str(interpolation_limit_time) + '_nans' + str(NaNs_percentage) + '.xlsx'
    
    # In all cases we log the output
    log_file, _ = os.path.splitext(output_file)
    log_file += "_ShipResampler_logfile.txt"
    
    # Rename output and log filename if file already exist
    output_file = Renamer(output_file).rename()
    log_file = Renamer(log_file).rename()

    sys.stdout = Logger(log_file, mode="w", verbose=args.verbose)

    # We display the input parameters
    print("--------- Input parameters: ---------------")
    print("-------------------------------------")
    pprint(params)

    # We apply eval on the aggregation functions specific to each variable if specified    
    # in the parameters. 
    for k,v in set_custom_agg_function.items():
        set_custom_agg_function[k] = eval(v)

    tps_debut_charg = time.perf_counter() 

    ########################################################################################
    # Captors data importation   #
    ##############################
    # Modification of sensor type data
    captors_files_list = []
    for captors_file_path, loaded_sheets in captors_files.items():
        captors_file = CaptorsFiles(captors_file_path,
                             loaded_sheets=loaded_sheets)
        captors_file.select_var(selected_variables=selected_variables)\
                    .var_with_too_much_nans_filter(NaNs_percentage)\
                    .merge_same_sheets(same_sheets=same_sheets)\
                    .strip_variables_names()\
                    .add_sheet_prefix()
        
        captors_files_list.append(captors_file)

    # Modification of VRS type data
    vrs_files_list = []              
    for vrs_file_path in vrs_files:
        vrs_file = VrsFiles(vrs_file_path)
        vrs_file.select_board(ship=name)\
                .select_var(selected_variables=selected_variables)\
                .var_with_too_much_nans_filter(NaNs_percentage)\
                .create_mean_draft()\
                .strip_variables_names()\
                .add_sheet_prefix()\
                .shift_date_var()
                                                                                           
        vrs_files_list.append(vrs_file)                                                    
    ########################################################################################  
    
    tps_fin_charg = time.perf_counter()

    # Creation of the resampler object with all the parameters
    tps_debut_exec = time.perf_counter() 
    ship_resampler_object = ShipResampler(
        files=captors_files_list + vrs_files_list,
        name=name,
        resampling_windows=pd.Timedelta(resampling_windows), 
        interpolation_method=interpolation_method,
        interpolation_limit_time=pd.Timedelta(interpolation_limit_time),
        agregation_function=eval(agregation_function),
        set_custom_agg_function=set_custom_agg_function, 
        set_custom_inter_method=set_custom_inter_method,
        set_custom_inter_limit_time=set_custom_inter_limit_time,
        geospatial_format=geospatial_format,
        dist_limit=dist_limit,
        agregation_string_function=agregation_string_function,
        fill_method_str=fill_method_str
        )
    
    ########################################################################################
    # Applying transformations to data   #
    ######################################
    ship_resampler_object.change_spatial_format(geospatial_format='dd')\
                    .change_deg_format(deg_format='continuous', include_geo_var=True)\
                    .resample_and_interpolate()\
                    .nans_filter()\
                    .distance_limit_filter()\
                    .change_deg_format(deg_format=deg_format, include_geo_var=True)\
                    .change_spatial_format(geospatial_format=geospatial_format)
    ######################################################################################## 
                    
    tps_fin_exec = time.perf_counter()
    
    # Saving pre-processed data
    tps_debut_enr = time.perf_counter() 
    ship_resampler_object.save_preprocessed_data(path=output_file)
    tps_fin_enr = time.perf_counter()
    
    # Display of execution times
    print('\n-------------------------------------')
    print('--------Execution times: ---------\n')
    print('Loading imports: ', tps_debut_charg-tps_debut, 's.')
    print('Loading input data: ', tps_fin_charg-tps_debut_charg, 's.')
    print('Execution of preprocessing script: ', tps_fin_exec-tps_debut_exec, 's.')
    print('Saving preprocessed data: ', tps_fin_enr-tps_debut_enr, 's.')
    
    return ship_resampler_object


if __name__ == "__main__":
    magellan_ship_resampler = main()
