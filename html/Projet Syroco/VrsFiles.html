# -*- coding: utf-8 -*-

# Imports
import numpy as np
import pandas as pd

# Class imports
from Files import Files


class VrsFiles(Files):
    """
    Class representing VRS files
    VrsFiles objects are considered as Files objects which have only one sheet named 'VRS'
    and have particular method to shift the date variable and select a board
    ...

    Attributes
    ----------
    same as Files object
    
    """
    
    def __init__(self, path, time_var=None, time_name='Date/Time', loaded_sheets=0):
        """
        Constructor of VRS files
        
        Parameters
        ----------
        path : string
            path of the file
            
        time_var : string, None
            name of the temporal variable or None for automatic detection 
            
        time_name : string
            for renaming the temporal variable
            
        loaded_sheets : List[string], None, int
            strings are used for sheet names, integers are used in zero-indexed sheet positions,
            None select all the sheets
              
        Returns
        -------
        """
        Files.__init__(self, path, time_var=time_var, time_name=time_name, loaded_sheets=loaded_sheets)
        self.rename_sheets({'feuille unique': 'VRS'})

    def shift_date_var(self, delay=pd.Timedelta(days=-1)):
        """
        Function which allows to shift the temporal variable
        of the VRS file
        
        Parameters
        ----------
        delay : pd.timedelta
              delay to add to the temporal variable
              
        Returns
        -------
        """
        self.content['VRS'][self.time_var] += delay
        
        return self
        
    def select_board(self, ship='CMA CGM MAGELLAN'):
        """
        Function that allows to select a ship in the VRS file
        
        Parameters
        ----------
        ship : string
            ship name
              
        Returns
        -------
        """
        self.content['VRS'] = self.content['VRS'].loc[self.content['VRS']['Vessel Name']==ship, :]
     
        return self
    
    def create_mean_draft(self):
        """
        Function which allows to create the variable 'mean_draft'
        
        Parameters
        ----------
        
        Returns
        -------
        """
        self.content['VRS']['Mean draft [m]'] = self.content['VRS']\
                                                     .loc[:, ['Draft FWD at PP (m) ', 'Draft AFT at PP (m)']]\
                                                     .apply(np.mean, axis=1)
        return self
