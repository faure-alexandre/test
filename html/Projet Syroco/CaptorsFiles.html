# -*- coding: utf-8 -*-

# Imports
import numpy as np
import pandas as pd

# Class imports
from Files import Files


class CaptorsFiles(Files):
    """
    Class representing onboard captors files
    CaptorsFiles objects are considered as Files objects which can have variables splited
    in differents sheets

    Attributes
    ----------
    same as Files object
    """
    
    def __init__(self, path, time_var=None, time_name='Date/Time', loaded_sheets=['BonVoyage', 'Bridge Sensor', 'Motion Sensor', 'MACS3', 'NAVIGATION_GPS', 'NMEA GPS',
                                      'NAVIGATION_GYRO', 'NMEA Gyro', 'NMEA_VDR1', 'NMEA_VDR2', 'NMEA_VDR3']):
        """
        Constructor of CaptorsFiles
        
        Parameters
        ----------
        path : string
            path of the file
            
        time_var : string, None
            name of the temporal variable or None for automatic detection 
            
        time_name : string
            for renaming the temporal variable
            
        loaded_sheets : List[string], None, int
            strings are used for sheet names, integers are used in zero-indexed sheet positions,
            None select all the sheets
            
        Returns
        -------
        
        """
        Files.__init__(self, path, time_var=time_var, time_name=time_name, loaded_sheets=loaded_sheets)
        
        
    def merge_same_sheets(self, same_sheets=[('NMEA Gyro','NAVIGATION_GYRO'), ('NMEA GPS','NAVIGATION_GPS'),
                       ('Bridge Sensor','Motion Sensor')]):
        
        """
        Function that allows to concatenate in the same sheet
        variables that are distributed in several different sheets
        
        Parameters
        ----------
        same_sheets : list[tuple(string)]
              sheets containing identical variables
              nb: the variables which are not present in all the sheets of the same
              pair are deleted
                                                        
        Returns
        -------
        """
        
        deleted_columns = []
        
        # Check if same_sheets sheets are in file
        print('\n-------------------------------------')
        print('Sheets deleted because one of the pairs was not loaded\n')
        taille = len(same_sheets)
        
        i = 0
        while i < taille:
            for j in range(len(same_sheets[i])):
                if same_sheets[i][j] not in self.sheets_name:
                    print(same_sheets[i])
                    del same_sheets[i]
                    taille -= 1
                    break
            i += 1
                
        for sheet_tuples in same_sheets:
            all_columns = []
            for sheet in sheet_tuples:
                all_columns += self.content[sheet].columns.tolist()
            # We keep only common columns
            common_columns = [ i for i in np.unique(all_columns) if all_columns.count(i) == len(sheet_tuples) ]
            
            merged_sheet = pd.DataFrame(columns=common_columns)
            for sheet in sheet_tuples:
                merged_sheet = pd.concat([merged_sheet, self.content[sheet][common_columns]], join='outer', ignore_index=True)
                
                if sheet != sheet_tuples[0]:
                    del self.content[sheet] 
            
            self.content[sheet_tuples[0]] = merged_sheet
            deleted_columns += [i for i in np.unique(all_columns) if i not in common_columns]
            
        self.var_names = {k: list(v.columns) for k, v in self.content.items()}
        self.sheets_names = list( self.content.keys() )
           
        # Display
        print('\n-------------------------------------')
        print('Variables deleted because only present in one sheet of 2 \'same sheet\'\n')
        
        return self
