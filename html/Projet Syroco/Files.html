# -*- coding: utf-8 -*-

# Imports
import pandas as pd
import os
from pprint import pprint
from lambda_utils import import_data_from_s3


class Files:
    """
    Generic class for the different temporal files (VRS and onboard captors data)
    ...
    
    Attributes
    ----------
    path : string
        path of the file
        
    filename : string
        name of the file
        
    extension : string
        extension of the file
        
    content : dict{string: DataFrame}
        each key correspond to a sheet name and value correspond to
        the content of the sheet
        
    time_var : string
        the name of the temporal variable
        
    sheets_name : list[string]
        list containing all sheet names
      
    var_names : dict{string: list[string]}
        dictionary containing for each key the name of the sheet and
        for each value the list of the variables names corresponding to the sheet
        
    self.var_types : dict{str: dict{str: list[str]}}
        dictionary containing for each key a sheet name and for value a dictionnary
        containing in key a type (keys: 'all_columns','string_columns','date_columns','non_num_columns',
        'num_columns') and in value a list containing the variables names corresponding to 
        this sheet and this type
    """
    
    
    def __init__(self, path, time_var, time_name, loaded_sheets=None):
        self.path = path
        self.filename, self.extension = os.path.splitext(path)

        """
        Function that loads data into a dictionary of dataframes
        each key corresponding to a sheet
        
        Parameters
        ----------
        path : string
            path of the file
            
        time_var : string, None
            name of the temporal variable or None for automatic detection 
            
        time_name : string
            for renaming the temporal variable
            
        loaded_sheets : List[string], None, int
            strings are used for sheet names, integers are used in zero-indexed sheet positions,
            None select all the sheets
            
        Returns
        -------
        
        """
        if self.path[:5] == 's3://':
            if self.extension == '.csv':
                self.content = pd.read_csv(import_data_from_s3(self.path))
            elif self.extension in ['.xlsx','.xls','.ods']:
                self.content = pd.read_excel(import_data_from_s3(self.path), sheet_name=loaded_sheets)
        else:
            if self.extension == '.csv':
                self.content = pd.read_csv(self.path)
            elif self.extension in ['.xlsx','.xls','.ods']:
                self.content = pd.read_excel(self.path, sheet_name=loaded_sheets)
                
        # Case where only one sheet is loaded
        if not isinstance(self.content, dict):
            content = self.content
            self.content = {}
            self.content['feuille unique'] = content
            
        self.sheets_name = list( self.content.keys() )
        self.var_names = {k: list(v.columns) for k, v in self.content.items()}
        self.load_columns_types()
        self.set_date_variable(time_var, time_name)
        
    def set_date_variable(self, var_name, new_var_name):
        """
        Function that loads the time variable and renames it
        
        Parameters
        ----------
        var_name : string, None
            name of the temporal variable or None for automatic detection 
            
        new_var_name : string
            new name of the temporal variable
            
        Returns
        -------
        
        """
        
        if var_name == None:
            for sheet_name, df in self.content.items():
                time_var = self.var_types[sheet_name]['date_columns'][0]
                self.content[sheet_name] = df.rename(columns={time_var: new_var_name})
        else:
            for sheet_name, df in self.content.items():
                time_var = var_name
                self.content[sheet_name] = df.rename(columns={time_var: new_var_name})
        
        self.time_var = new_var_name
        
    def strip_variables_names(self):
        """
        Function that removes left and right spaces in the variables names
        
        Parameters
        ----------
            
        Returns
        -------
        
        """
        for sheet_name, df in self.content.items():
            self.content[sheet_name].columns = df.columns.str.strip()
            
        self.var_names = {k: list(v.columns) for k, v in self.content.items()}
        
        return self
    
    
    def select_var(self, selected_variables):
        """
        Function that allows to select only certain variables
        
        Parameters
        ----------
        selected_variables : dict{string: list[string]}
            dictionary containing in key the name of the sheet and in value a list
            containing the name of the variables to keep. If a sheet is not present in 
            selected_variables, all the variables for this sheet are kept
            
        Returns
        -------
        """
    
        for sheet_name, df in self.content.items():
            if sheet_name in selected_variables.keys():
                self.content[sheet_name] = df[selected_variables[sheet_name]]
      
        return self

      
    def add_sheet_prefix(self):
        """
        Function which allows to add in prefix of each variable
        the name of the sheet
        
        Parameters
        ----------
               
        Returns
        -------
        """
        for sheet_name, df in self.content.items():
            self.content[sheet_name] = df.add_prefix(sheet_name+'_')\
                                        .rename(columns={(sheet_name+'_'+self.time_var): self.time_var})
    
        self.var_names = {k: list(v.columns) for k, v in self.content.items()}
        
        return self
    
    def merge_all_sheets(self):
        """
        Function that allows to concat all the sheets in one
        DataFrame
        
        Parameters
        ----------
               
        Returns
        -------
        merged_content : DataFrame
            DataFrame containing all concatenated sheets
        """
        merged_content = pd.concat(self.content, join='outer', ignore_index=True)
        
        return merged_content
        
        
    def print_file_infos(self, sheet_name, nb_rows):
        """
        Function that displays the content of the file
        
        Parameters
        ----------
        sheet_name : list[string]
            list containing all the sheets to display
            
        nb_rows : int
            number of rows to display for each sheet
            
        Returns
        -------
        """
        print('----------------------------------------------')
        print('------------', self.filename, '---------------\n')
        print('Sheet number: ', len(self.sheets_name))
        print('Sheet names: ', self.sheets_name)
        
        for sheet in sheet_name:
            print('\n-------------------------------------------')
            print('------------', sheet, '-----------\n')
            print(self.content[sheet].head(nb_rows))

        
    def rename_sheets(self, new_sheet_names):
        """
        Function that allows to rename certain sheets
        
        Parameters
        ----------
        new_sheet_names : dict{string: string}
           dictionary containing in key the old name of sheet and in value the new name
            
        Returns
        -------
        """
        for k, v in new_sheet_names.items():
            self.content[v] = self.content.pop(k)

        self.var_names = {k: list(v.columns) for k, v in self.content.items()}
        self.sheets_names = list( self.content.keys() )
        
        return self
        
    def get_sheets_stats(self):
        """
        Function which allows to calculate the basic statistics (average, minimum, standard deviation...)
        for each variable of each sheet
        
        Parameters
        ----------
        
        Returns
        -------
        sheets_stats : dict(string: DataFrame)
            dictionary containing in key the name of the sheet and in value
            the DataFrame containing the stats of the sheet
        """
        sheets_stats = {}
        
        for sheet_name, df in self.content.items():
            sheets_stats[sheet_name] = df.describe()
            
        return sheets_stats
        
    
    def load_columns_types(self):
        """
        Function that allows you to create dictionary containing for each sheet
        the type of each variable
        
        Parameters
        ----------
        
        Returns
        -------
        """
        sheets_var_types = {}
        
        for sheet_name, df in self.content.items():
            sheets_var_types[sheet_name] = {}
        
            # Locate columns containing str or date values, and the others that are supposed numeric 
            sheets_var_types[sheet_name]['all_columns'] = df.columns.tolist()
            sheets_var_types[sheet_name]['string_columns'] = df.dtypes[df.dtypes == 'object'].index.tolist()
            sheets_var_types[sheet_name]['date_columns'] = df.dtypes[df.dtypes == 'datetime64[ns]'].index.tolist()
            sheets_var_types[sheet_name]['non_num_columns'] = sheets_var_types[sheet_name]['date_columns'] + sheets_var_types[sheet_name]['string_columns']
            sheets_var_types[sheet_name]['num_columns'] = [col for col in df.columns.tolist() if col not in sheets_var_types[sheet_name]['non_num_columns'] ]
            
        self.var_types = sheets_var_types
            
    
    def var_with_too_much_nans_filter(self, NaNs_percentage):
        """
        Function which allows to delete variables containing too much
        NaNs.
        
        Parameters
        ----------
        nans_percentage : float
            NaNs percentage tolerated for each variable
            
        Returns
        -------
        """
        deleted_variables = []
        
        for sheet_name, df in self.content.items():
            deleted_variables += list(df.columns[df.isna().sum()/df.shape[0] >= NaNs_percentage])
            self.content[sheet_name] = df.loc[:, df.columns[df.isna().sum()/df.shape[0] < NaNs_percentage]]
       
        print('\n-------------------------------------')
        print('Deleted variables because proportion of NaNs is greater than ', NaNs_percentage*100, '%')
        print('in the file: ', self.filename, '\n')
        pprint(deleted_variables)
        
        return self
    
        
    def save_file(self, path):
        """
        Function which allows to save the object in an .xlsx file
        
        Parameters
        ----------
        path : string
            path name of the .xlsx file
            
        Returns
        -------
        """
        
        writer = pd.ExcelWriter(path)
        
        for sheet_name, df in self.content.items():
            # Write each dataframe to a different worksheet.
            df.to_excel(writer, sheet_name=sheet_name)

        # Close the Pandas Excel writer and output the Excel file.
        writer.save()
        
        return self

def set_date_variable(self, var_name, new_var_name):
    """
    Function that loads the time variable and renames it
    
    Parameters
    ----------
    var_name : string, None
        name of the temporal variable or None for automatic detection 
        
    new_var_name : string
        new name of the temporal variable
        
    Returns
    -------
    
    """
    
    if var_name == None:
        for sheet_name, df in self.content.items():
            time_var = self.var_types[sheet_name]['date_columns'][0]
            self.content[sheet_name] = df.rename(columns={time_var: new_var_name})
    else:
        for sheet_name, df in self.content.items():
            time_var = var_name
            self.content[sheet_name] = df.rename(columns={time_var: new_var_name})
    
    self.time_var = new_var_name

def var_with_too_much_nans_filter(self, NaNs_percentage):
    """
    Function which allows to delete variables containing too much
    NaNs.
    
    Parameters
    ----------
    nans_percentage : float
        NaNs percentage tolerated for each variable
        
    Returns
    -------
    """
    deleted_variables = []
    
    for sheet_name, df in self.content.items():
        deleted_variables += list(df.columns[df.isna().sum()/df.shape[0] >= NaNs_percentage])
        self.content[sheet_name] = df.loc[:, df.columns[df.isna().sum()/df.shape[0] < NaNs_percentage]]
   
    print('\n-------------------------------------')
    print('Deleted variables because proportion of NaNs is greater than ', NaNs_percentage*100, '%')
    print('in the file: ', self.filename, '\n')
    pprint(deleted_variables)
    
    return self
