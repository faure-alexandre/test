# -*- coding: utf-8 -*-

import openpyxl
import pandas as pd
import boto3
import io
import os
import aws_config

import pymysql
import sys
import logging
from sqlalchemy import create_engine
from Renamer import Renamer
    

def import_dataframe_in_RDS(df, table_name):
    """
    Function allowing to save a dataframe in the AWS RDS database
    
    Parameters:
    -----------
        df : pd.DataFrame
            DataFrame to save in RDS
        
        table_name : str
            table name to write in RDS
    """
    rds_host  = aws_config.db_endpoint
    name = aws_config.db_username
    password = aws_config.db_password
    db_name = aws_config.db_name
    
    logger = logging.getLogger()
    logger.setLevel(logging.INFO)
    
    # Connection to RDS bdd
    try:
        conn = pymysql.connect(host=rds_host, user=name, passwd=password, connect_timeout=5)
    except pymysql.MySQLError as e:
        logger.error("ERROR: Unexpected error: Could not connect to MySQL instance.")
        logger.error(e)
        sys.exit()
    
    logger.info("SUCCESS: Connection to RDS MySQL instance succeeded")

    # Modification of the RDS bdd
    with conn.cursor() as cur:
        cur.execute("CREATE DATABASE IF NOT EXISTS `{}`;".format(db_name))
        cur.execute("USE `{}`;".format(db_name))
        logger.info("BDD créée !")

        logger.info("Initial table name : " + table_name)
        # Rename the table name if it already exists in the RDS database
        new_table_name = Renamer(table_name).rename_table(cur)
        logger.info("Renamed table : " + new_table_name)

        cur.close()
        conn.close()
        
    # Creation of an SQLAlchemy engine
    engine = create_engine("mysql+pymysql://{user}:{pwd}@{host}/{db}".format(
                                        user=name,
                                        pwd=password,
                                        host=rds_host,
                                        db=db_name))
    logger.info('Connected to SQLAlchemy !')
    
    # Saving DataFrame in RDS
    df.to_sql(new_table_name, con=engine, if_exists='replace', index=False)
    
    logger.info("Dataframe saved in RDS !")
        
    return new_table_name
    
    
def import_data_from_s3(file_path):
    """
    Function allowing to import binary content of a file saved in AWS S3
    
    Parameters:
    -----------
        file_path : string
            S3 path of the file
            
    Returns:
    -----------
        string
            binary content of the s3 file
        
    """
    l = file_path.split('/')
    bucket_name = l[2]
    file_name = l[3] + '/' + l[4]
    s3_client = boto3.client('s3')
    s3_obj = s3_client.get_object(Bucket=bucket_name, Key=file_name)
    
    return io.BytesIO(s3_obj['Body'].read())
